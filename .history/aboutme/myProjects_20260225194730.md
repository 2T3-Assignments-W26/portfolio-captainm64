# Final Game Project: Get that @!!!!

This was my final game project for Programming 1, and the way it works is that you control a "*" symbol, and you have to find and touch an "@" before the timer runs out.

## Technologies Used 

Visual Studio 2022

## Code Snippets

# The main Draw function
             void Draw()
            {

                Console.SetCursorPosition(OldMeX, OldMeY);
                Console.Write(Empty);
                Console.SetCursorPosition(MeX, MeY);
                Console.Write(Player);
                Console.SetCursorPosition(CoinX, CoinY);
                Console.Write(Coin);
                Console.SetCursorPosition(Console.WindowTop, Console.WindowLeft);
                Console.Write(FormatHUD(Score, Timer));
            }

 # Main movement code

             void Move(ConsoleKey Inputt)
            {
                // Movement code that respects the console boundaries, parts of it found were online

                OldMeX = MeX;
                OldMeY = MeY;
                int minY = 1;
                int maxY = Console.WindowHeight - 1;
                int minX = 0;
                int maxX = Console.WindowWidth - 1;


                if (Inputt == ConsoleKey.UpArrow)
                {
                    if (MeY > minY) MeY--;
                }
                if (Inputt == ConsoleKey.W)
                {
                    if (MeY > minY) MeY--;
                }
                else if (Inputt == ConsoleKey.DownArrow)
                {
                    if (MeY < maxY) MeY++;
                }
                else if (Inputt == ConsoleKey.S)
                {
                    if (MeY < maxY) MeY++;
                }
                else if (Inputt == ConsoleKey.LeftArrow)
                {
                    if (MeX > minX) MeX--;
                }
                else if (Inputt == ConsoleKey.A)
                {
                    if (MeX > minX) MeX--;
                }
                else if (Inputt == ConsoleKey.RightArrow)
                {
                    if (MeX < maxX) MeX++;
                }
                else if (Inputt == ConsoleKey.D)
                {
                    if (MeX < maxX) MeX++;
                }
            }
